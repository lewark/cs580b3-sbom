Commands to run for the demo:

0. Setup

./setup.sh

1. Run Ollama

cd ~/cs580b3/ollama/
./bin/ollama serve

2. Run Chroma

source venv/bin/activate
./run-chroma.sh

3. Run agent

source venv/bin/activate
rm -r logs
./run-grid.sh --tooling --podman

4. Parse log

python3 -m sbom.logs.parser logs_short/tomcat/iteration1/

5. Run LLM-J

Set OLLAMA_MODEL = 'gpt-oss:20b-cloud' in analyze_logs.py

export NVD_DIR=../nvd VULNRICHMENT_DIR=../vulnrichment
source venv/bin/activate
python3 -m sbom.llm_j.analyze_logs logs/parsed-logs/

6. Aggregate results

rm -r figures
source venv/bin/activate
python3 -m sbom.llm_j.aggregate_results logs_full/llm-j-analysis-logs

7. Confusion matrix

export NVD_DIR=../nvd VULNRICHMENT_DIR=../vulnrichment
source venv/bin/activate
python3 -m sbom.llm_j.confusion_matrix logs_full
